1.DDL（数据定义语言）
CREATE DATABASE - 创建新数据库
ALTER DATABASE - 修改数据库
CREATE TABLE - 创建新表
ALTER TABLE - 变更（改变）数据库表
DROP TABLE - 删除表
CREATE INDEX - 创建索引（搜索键）
DROP INDEX - 删除索引
2.DML 数据操作语言
SELECT - 从数据库表中获取数据
UPDATE - 更新数据库表中的数据
DELETE - 从数据库表中删除数据
INSERT INTO - 向数据库表中插入数据

//表
永久表（create），临时表(查询)，虚拟表(CREATE VIEW tab AS condition:P46, 隐藏列，简化数据库)

// 查询: SELECT, FROM, WHERE, GROUP BY, HAVING, ORDER BY
SELECT cols (*/col as otherName/DISTINCT col：去重)
FROM table1 AS otherName INTER JOIN table2 ON condition // 连接表
WHERE condition AND condition (OR， IS, NOT 最好在condition中用!表示, IN, NOT IN) // BETWEEN AND // attr != ALL (select ... 集合)
OREDR BY col DESC;  // ASC 默认升序 // 排序后不能再group by组合

// 插入
INSERT INTO table (args)
VALUES (vals);
// 更新
UPDATE table
SET key = val
WHREE condition;
// 删除
DELETE FROM person
WHERE condition

获取xml格式的数据
mysql -u user -p --xml basename
不同 ：在每个查询末尾添加 for xml 子句即可
SELECT * FROM favorite_food
FOR XML AUTO, ELEMENTS

错误提示:
主键不唯一
不存在的外键
列值不合法
无效的日期转换
	使用str_to_date函数

where:
通配符：LIKE
_ 1个字符
% 任意数目的字符
F%  以F打头的字符串
%t  以t结尾的字符串
%bas%  包含'bas'字符串的字符串
__t_ 包含4个字符，且第三个为t的字符串

正则表达式: REGEXP '^[FG]' // 以F或G开头的字符串
IS NULL // 判断空
